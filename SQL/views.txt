CREATE OR REPLACE VIEW vw_sales_summary AS
SELECT
    s.sales_id,
    s.sales_date,
    s.payment_type,
    SUM(si.line_total) AS revenue,
    SUM(si.quantity * p.buying_price) AS cogs,
    SUM(si.line_total) - SUM(si.quantity * p.buying_price) AS net_profit
FROM sales_header s
JOIN sales_items si ON s.sales_id = si.sales_id
JOIN products p ON si.product_id = p.product_id
GROUP BY s.sales_id, s.sales_date, s.payment_type;

CREATE OR REPLACE VIEW vw_aov_gross_margin AS
SELECT
	TO_CHAR(DATE_TRUNC('month', s.sales_date), 'YYYY-MM') AS month,
    COUNT(DISTINCT s.sales_id) AS total_orders,
    SUM(si.line_total) / NULLIF(COUNT(DISTINCT s.sales_id),0) AS avg_order_value,
    (SUM(si.line_total) - SUM(si.quantity * p.buying_price)) / NULLIF(SUM(si.line_total),0) * 100 AS gross_margin_percent
FROM sales_header s
JOIN sales_items si ON s.sales_id = si.sales_id
JOIN products p ON si.product_id = p.product_id
GROUP BY TO_CHAR(DATE_TRUNC('month', s.sales_date), 'YYYY-MM');

CREATE OR REPLACE VIEW vw_revenue_trends_daily AS
SELECT
    s.sales_date,
    SUM(si.line_total) AS revenue,
    SUM(si.quantity * p.buying_price) AS cogs,
    SUM(si.line_total) - SUM(si.quantity * p.buying_price) AS net_profit
FROM sales_header s
JOIN sales_items si ON s.sales_id = si.sales_id
JOIN products p ON si.product_id = p.product_id
GROUP BY s.sales_date
ORDER BY s.sales_date;

CREATE OR REPLACE VIEW vw_revenue_trends_monthly AS
SELECT
    TO_CHAR(DATE_TRUNC('month', s.sales_date), 'YYYY-MM') AS month,
    SUM(si.line_total) AS revenue,
    SUM(si.quantity * p.buying_price) AS cogs,
    SUM(si.line_total) - SUM(si.quantity * p.buying_price) AS net_profit
FROM sales_header s
JOIN sales_items si ON s.sales_id = si.sales_id
JOIN products p ON si.product_id = p.product_id
GROUP BY DATE_TRUNC('month', s.sales_date)
ORDER BY month;

CREATE OR REPLACE VIEW vw_payment_mix AS
SELECT
    s.sales_date,
    payment_type,
    SUM(total_amount) AS total_amount,
    SUM(total_amount) * 100.0 / NULLIF(SUM(SUM(total_amount)) OVER (),0) AS percent_of_total
FROM sales_header
GROUP BY s.sales_date, payment_type;

CREATE OR REPLACE VIEW vw_product_performance AS
SELECT
    TO_CHAR(DATE_TRUNC('month', si.sales_date), 'YYYY-MM') AS month,
    p.product_id,
    p.product_name,
    p.category,
    SUM(si.line_total) AS revenue,
	SUM(si.quantity) AS quantity,
    SUM(si.quantity * p.buying_price) AS cogs,
    SUM(si.line_total) - SUM(si.quantity * p.buying_price) AS profit,
    CASE 
        WHEN SUM(si.line_total) > 0 THEN
            (SUM(si.line_total) - SUM(si.quantity * p.buying_price)) 
            / SUM(si.line_total) * 100
        ELSE 0 
    END AS margin_percent
FROM sales_items si 
JOIN products p ON si.product_id = p.product_id
GROUP BY 
    TO_CHAR(DATE_TRUNC('month', si.sales_date), 'YYYY-MM'),
    p.product_id,
    p.product_name,
    p.category;

CREATE OR REPLACE VIEW vw_top_bottom_products AS
SELECT *
FROM vw_product_performance
ORDER BY profit DESC
LIMIT 5

UNION ALL

SELECT *
FROM vw_product_performance
ORDER BY profit ASC
LIMIT 5;

CREATE OR REPLACE VIEW vw_inventory_aging_summary AS
SELECT
    p.product_id,
    p.product_name,
    SUM(i.remaining_qty) AS current_stock,
    SUM(CASE WHEN i.stock_age_days > 90 THEN i.remaining_qty ELSE 0 END) AS dead_stock,
    SUM(i.remaining_qty) - SUM(CASE WHEN i.stock_age_days > 90 THEN i.remaining_qty ELSE 0 END) AS usable_stock
FROM inventory_aging i
JOIN products p ON i.product_id = p.product_id
GROUP BY p.product_id, p.product_name;

CREATE VIEW vw_vendor_payables AS
SELECT 
    v.vendor_id,
    v.vendor_name,
    SUM(po.remaining_amount) AS total_payables
FROM purchase_orders po
JOIN vendors v ON po.vendor_id = v.vendor_id
GROUP BY v.vendor_id, v.vendor_name;

CREATE VIEW vw_vendor_payment_performance AS
SELECT 
    v.vendor_id,
    AVG(p.payment_date - p.due_date) AS avg_delay,
    (COUNT(*) FILTER (WHERE p.payment_date > p.due_date) * 100.0) 
        / COUNT(*) AS late_percentage
FROM purchase_order_payment p
JOIN purchase_orders po ON p.purchase_order_id = po.purchase_order_id
JOIN vendors v ON po.vendor_id = v.vendor_id
WHERE p.payment_date IS NOT NULL
GROUP BY v.vendor_id;

CREATE VIEW vw_vendor_profit_contribution AS
SELECT
    v.vendor_id,
    v.vendor_name,
    SUM((si.selling_price - p.buying_price) * si.quantity) AS total_profit,
    SUM(si.selling_price * si.quantity) AS total_revenue,
    SUM(si.quantity) AS total_quantity
FROM sales_items si
JOIN products p
    ON si.product_id = p.product_id
JOIN vendors v
    ON p.vendor_id = v.vendor_id
GROUP BY v.vendor_id, v.vendor_name;

CREATE VIEW vw_inventory_stock_status AS
SELECT 
    p.product_id,
    p.product_name,
    p.category,
    ps.usable_stock,
    rl.reorder_level,
    ps.inventory_value,

    CASE
        WHEN ps.usable_stock = 0 THEN 'OUT_OF_STOCK'
        WHEN ps.usable_stock <= rl.reorder_level THEN 'LOW_STOCK'
        WHEN ps.usable_stock > rl.reorder_level * 2 THEN 'OVERSTOCK'
        ELSE 'HEALTHY'
    END AS stock_status

FROM products p
JOIN product_summary ps ON p.product_id = ps.product_id
LEFT JOIN reorder_level rl ON p.product_id = rl.product_id;

CREATE VIEW vw_inventory_category_value AS
SELECT
    p.category,
    SUM(ps.inventory_value) AS inventory_value,
    COUNT(p.product_id) AS product_count
FROM products p
JOIN product_summary ps 
    ON p.product_id = ps.product_id
GROUP BY p.category;

CREATE OR REPLACE VIEW vw_inventory_health_metrics AS
SELECT

    inv.total_inventory_value,
    inv.overstock_value,
    inv.dead_stock_value,
    cogs.total_cogs

FROM
(
    SELECT

        SUM(ps.inventory_value) AS total_inventory_value,

        SUM(CASE
            WHEN iss.stock_status = 'OVERSTOCK'
            THEN ps.inventory_value ELSE 0 END) AS overstock_value,

        SUM(CASE
            WHEN ia.is_dead_stock = TRUE
            THEN ia.remaining_qty * ia.buying_price ELSE 0 END) AS dead_stock_value

    FROM product_summary ps

    LEFT JOIN vw_inventory_stock_status iss
        ON ps.product_id = iss.product_id

    LEFT JOIN inventory_aging ia
        ON ps.product_id = ia.product_id

) inv

CROSS JOIN
(
    SELECT
        SUM(si.quantity * p.buying_price) AS total_cogs
    FROM sales_items si
    JOIN products p
        ON si.product_id = p.product_id
    WHERE si.sales_date BETWEEN '2020-01-01' AND '2020-12-29'
) cogs;

CREATE VIEW vw_product_velocity AS
SELECT

    p.product_id,
    p.product_name,

    SUM(si.quantity) AS total_sold,

    ps.current_stock,

    (SUM(si.quantity) / NULLIF(ps.current_stock + SUM(si.quantity),0))
        AS sell_through_rate

FROM products p

JOIN sales_items si 
    ON p.product_id = si.product_id

JOIN product_summary ps
    ON p.product_id = ps.product_id

GROUP BY
    p.product_id,
    p.product_name,
    ps.current_stock;
	

CREATE OR REPLACE VIEW vw_inventory_risk_products AS
SELECT *
FROM (
    SELECT

        p.product_id,
        p.product_name,
        p.category,
        ps.usable_stock,
        ps.inventory_value,

        CASE
            WHEN ps.usable_stock = 0 THEN 'OUT_OF_STOCK'
            WHEN ps.usable_stock <= rl.reorder_level THEN 'LOW_STOCK'
            WHEN ps.usable_stock > rl.reorder_level * 2 THEN 'OVERSTOCK'
        END AS risk_type

    FROM products p
    JOIN product_summary ps
        ON p.product_id = ps.product_id
    LEFT JOIN reorder_level rl
        ON p.product_id = rl.product_id
) t

WHERE risk_type IS NOT NULL;
